//Kdekoli v hlavnim tele programu se muze vyskytnout statement nebo definice fce
<program> -> <statement> <program>
<program> -> <def_function> <program>
<program> -> EOF

//term_n je potreba pro volani fce kde muze byt vicekrat za sebou oddelen carkou
<term> -> value <term_n>
<term> -> id <term_n>
<term> -> None <term_n>
<term> -> Ɛ
<term_n> -> , <term> 
<term_n> -> Ɛ 

//function definitions
<def_function> -> def id ( <param> ) : EOL INDENT <statement> <return> <statement> DEDENT
<param> -> id <param_n>
<param> -> Ɛ
<param_n> -> , <param>
<param_n> -> Ɛ

<return> -> return <return_value> EOL
<return> -> Ɛ
<return_value> -> <expression>
<return_value> -> Ɛ 

//prikazy ktere se mohou vyskytnout kdekoli v tele programu
<statement> -> <expression> EOL <statement>
<statement> -> id <idwhat> EOL <statement>
<statement> -> if <expression> : EOL INDENT <statement> DEDENT else : EOL INDENT <statement> DEDENT <statement>
<statement> -> while <expression> : EOL INDENT <statement> DEDENT <statement>
<statement> -> pass EOL <statement>
<statement> -> Ɛ

//id muze byt bud promenna nebo nazev fce
<idwhat> -> = <assign>
<idwhat> -> ( <term> )

//do promenne muze byt prirazen vysledek volane fce nebo hodnota vyrazu(prirazeni jine samostatne prom. se vyhodnocuje jako vyraz)
<assign> -> id( <term> )
<assign> -> <expression>
