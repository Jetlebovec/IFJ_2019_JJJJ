//Kdekoli v hlavnim tele programu se muze vyskytnout statement nebo definice fce
0. <program> -> EOL <program>
1. <program> -> <statement> <program>
2. <program> -> <def_function> <program>
3. <program> -> EOF

//term_n je potreba pro volani fce kde muze byt vicekrat za sebou oddelen carkou
4. <term> -> value <term_n>
5. <term> -> id <term_n>
6. <term> -> None <term_n>
7. <term> -> Ɛ
8. <term_n> -> , <term> 
9. <term_n> -> Ɛ 

//function definitions
10. <def_function> -> def id ( <param> ) : EOL INDENT <statement> <return> <statement> DEDENT
11. <param> -> id <param_n>
12. <param> -> Ɛ
13. <param_n> -> , <param>
14. <param_n> -> Ɛ

15. <return> -> return <return_value> EOL
16. <return> -> Ɛ
17. <return_value> -> <expression>
18. <return_value> -> Ɛ 

//prikazy ktere se mohou vyskytnout kdekoli v tele programu
19. <statement> -> <expression> EOL <statement>
20. <statement> -> id <idwhat> EOL <statement>
21. <statement> -> if <expression> : EOL INDENT <statement> DEDENT else : EOL INDENT <statement> DEDENT <statement>
22. <statement> -> while <expression> : EOL INDENT <statement> DEDENT <statement>
23. <statement> -> pass EOL <statement>
24. <statement> -> Ɛ

//id muze byt bud promenna nebo nazev fce
25. <idwhat> -> = <assign>
26. <idwhat> -> ( <term> )

//do promenne muze byt prirazen vysledek volane fce nebo hodnota vyrazu(prirazeni jine samostatne prom. se vyhodnocuje jako vyraz)
27. <assign> -> id( <term> )
28. <assign> -> <expression>
